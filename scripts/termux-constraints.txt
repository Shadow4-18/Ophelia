# Termux / Android pip constraints for Ophelia.
#
# Termux is Linux userland — PEP 508 cannot target it. PyPI manylinux wheels
# often do not match Termux's libc, so pip falls back to Rust/C source builds
# (maturin) which fail without ANDROID_API_LEVEL + a full NDK toolchain.
#
# Usage (from repo root):
#   source scripts/termux-pip-env.sh
#   termux_preinstall_native_wheels
#   termux_pip_install -e . -c scripts/termux-constraints.txt
#
# openai>=1.40 depends on jiter (Rust) — no reliable Termux wheel on PyPI.
openai>=1.35,<1.40
# openai<1.40 passes proxies= to httpx; removed in httpx 0.28+.
httpx>=0.27,<0.28
